bool prime(int a)
{
    if (a < 2) return false;
    if (a == 2 || a == 3) return true;
    if (!(a & 0x01)) return false;
    int m = floor(sqrt((double)a) + 0.5);
    //cout << a << ":" << m << endl;
    for (int i = 3; i <= m; i += 2) {
        if (a % i == 0) return false;
    }
    return true;
}

int main()
{
    int n;
    int a, count;
    while (cin >> n) {
        count = 0;
        while (n--) {
            cin >> a;
            if (prime(a)) count++;
        }
        cout << count << endl;
    }

    return 0;
}
