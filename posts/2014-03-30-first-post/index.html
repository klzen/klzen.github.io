// 2014年 3月29日 星期六 08时31分21秒 CST
#include <iostream>
#include <algorithm>
#include <numeric>
#include <bitset>
#include <vector>
#include <stack>
#include <cstdio>
#include <cstring>
#include <cmath>

using namespace std;

#define fe(i, a, b) for (int i = int(a); i <= int(b); i++)
#define fu(i, a, b) for (int i = int(a); i < int(b); i++)
typedef vector<int> vi;
typedef vector<vi> vvi;
typedef vector<bool> vb;
typedef vector<vb> vvb;
typedef pair<int, int> pii;
#define pb push_back
#define mp make_pair

struct M {
    int pi, mi;
    M(int p, int m) : pi(p), mi(m) {}
    bool operator<(const M another) const {
        return pi > another.pi;
    }
};

int main()
{
    int v, n, p, m;
    vector<M> vm;
    int value;
    while (true) {
        vm.clear();
        value   = 0;
        cin >> v;
        if (!v) break;
        cin >> n;
        while (n--) {
            cin >> p >> m;
             vm.pb(M(p, m));
        }
        sort(vm.begin(), vm.end());
        for (int i = 0; i < vm.size(); i++) {
            //cout << vm[i].pi << endl;
            if (v >= vm[i].mi) {
                value += vm[i].pi * vm[i].mi;
                v -= vm[i].mi;
            } else {
                value += vm[i].pi * v;
                break;
            }
        }
        cout << value << endl;
    }

    return 0;
}
